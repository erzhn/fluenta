### Шаг 1: Создай функцию getNextLesson в lib/lessons-data.ts

Добавь в конец файла:

```ts
import { getLessonCompletions } from '@/lib/progress'

export async function getNextLesson(userId: string | null): Promise<Lesson | null> {
  // Берём все завершённые уроки
  const completions = userId
    ? await getLessonCompletions(userId)
    : (() => {
        try {
          const raw = localStorage.getItem('fluenta_lesson_progress')
          const data = raw ? JSON.parse(raw) : {}
          return data.completedLessons ?? {}
        } catch { return {} }
      })()

  const completedIds = new Set(Object.keys(completions))

  // Находим первый незавершённый урок в порядке уровней и блоков
  const levelOrder = ['A1', 'A2', 'B1', 'B2', 'C1']
  for (const level of levelOrder) {
    const levelLessons = LESSONS
      .filter(l => l.level === level)
      .sort((a, b) => a.block - b.block || a.order - b.order)
    for (const lesson of levelLessons) {
      if (!completedIds.has(lesson.id)) return lesson
    }
  }
  return null
}

export function getLastCompletedLesson(completedIds: Set<string>): Lesson | null {
  const levelOrder = ['C1', 'B2', 'B1', 'A2', 'A1']
  for (const level of levelOrder) {
    const levelLessons = LESSONS
      .filter(l => l.level === level && completedIds.has(l.id))
      .sort((a, b) => b.block - a.block || b.order - a.order)
    if (levelLessons.length > 0) return levelLessons[0]
  }
  return null
}
```

```ts
// Найти следующий урок
async function loadNextLesson(uid: string | null) {
  setNextLessonLoading(true)
  try {
    const completions = uid
      ? await getLessonCompletions(uid)
      : (() => {
          try {
            const raw = localStorage.getItem('fluenta_lesson_progress')
            const data = raw ? JSON.parse(raw) : {}
            return data.completedLessons ?? {}
          } catch { return {} }
        })()

    const completedIds = new Set(Object.keys(completions))
    setCompletedCount(completedIds.size)

    const levelOrder = ['A1', 'A2', 'B1', 'B2', 'C1']
    let found: import('@/lib/lessons-data').Lesson | null = null
    for (const level of levelOrder) {
      const levelLessons = LESSONS
        .filter(l => l.level === level)
        .sort((a, b) => a.block - b.block || a.order - b.order)
      for (const lesson of levelLessons) {
        if (!completedIds.has(lesson.id)) { found = lesson; break }
      }
      if (found) break
    }
    setNextLesson(found)
  } finally {
    setNextLessonLoading(false)
  }
}
loadNextLesson(userId)  // вызови в useEffect когда userId известен
```
